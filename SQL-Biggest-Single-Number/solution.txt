Table: MyNumbers

+-------------+------+
| Column Name | Type |
+-------------+------+
| num         | int  |
+-------------+------+
This table may contain duplicates (In other words, there is no primary key for this table in SQL).
Each row of this table contains an integer.

 

A single number is a number that appeared only once in the MyNumbers table.

Find the largest single number. If there is no single number, report null.

# Write your MySQL query statement below

with newtable as (
select num, count(num) as ct from mynumbers
group by num
having count(num)=1
)
select distinct
    case when count(num)>0 then (select num from newtable order by num desc limit 1)
    else null
    end as num
    from newtable

select max(num) as num
from (select num 
      from MyNumbers
      group by num
      having count(num) = 1) as tbl

WITH SingleOccurrence AS (
    SELECT num
    FROM mynumbers
    GROUP BY num
    HAVING COUNT(*) = 1
)
SELECT 
    IFNULL(
        (SELECT num FROM SingleOccurrence ORDER BY num DESC LIMIT 1),
        NULL
    ) AS result;
